// Simulação de filtro de vagas
function filtrarVagas() {
    const localizacao = document.getElementById('localizacao').value.toLowerCase();
    const tipo = document.getElementById('tipo').value;
    const vagas = document.querySelectorAll('.vaga');

    vagas.forEach(vaga => {
        const texto = vaga.textContent.toLowerCase();
        const matchLocal = !localizacao || texto.includes(localizacao);
        const matchTipo = !tipo || texto.includes(tipo);
        vaga.style.display = matchLocal && matchTipo ? 'block' : 'none';
    });
}
